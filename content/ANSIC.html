    printf("ax + ux = %lu\n", ax + ux);

    return 0;
}

#include <stdio.h>
int main()
{
    int days, years, weeks;

    days = 1329; // 總天數

    // 將天數轉換成年、週和剩餘的天數
    years = days / 365; // 計算年數
    weeks = (days % 365) / 7; // 計算週數
    days = days - ((years * 365) + (weeks * 7)); // 計算剩餘的天數

    // 印出結果
    printf("年數: %d\n", years);
    printf("週數: %d\n", weeks);
    printf("天數: %d \n", days);

    return 0;
}
